const QRCode = require('qrcode');

// 容錯率 L &lt; M &lt; Q &lt; H
const opts = {
    errorCorrectionLevel: 'H',
    version: 2
};
//content
//url : const text = 'www.google.com';
const text = 'Hello Marc!!';
//path
const path = './qrcode.png';
QRCode.toFile(path, text, opts, (err) => {
    if (err) throw err;
    console.log('saved.');
});

const express = require('express')
const app = express()
const QRCode = require('qrcode');

app.get('/:qrcode',(req, res) =>{

    let inputStr = req.params.qrcode;

    QRCode.toDataURL(inputStr, function (err, url) {


        let data = url.replace(/.*,/,'')
        let img = new Buffer(data,'base64')
        res.writeHead(200,{
            'Content-Type' : 'image/png',
            'Content-Length' : img.length
        })
        res.end(img)

    })


})

app.listen(3000)
